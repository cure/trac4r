    def self.load params
      ticket = self.new params[0]
      ticket.created_at = params[1]
      ticket.updated_at = params[2]
      attributes = params[3]
      attributes.each do |key,value|
        ticket.instance_variable_set("@#{key}".to_sym,value)
      end
      return ticket
    end
